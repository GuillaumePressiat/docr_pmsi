pmsi_rust en appel dans un programme rust

Exemple de lecture d’un fichier ano 2022

let file_to_read = "/Users/guillaumepressiat/Documents/data/mco/290000017.2022.12.ano";
let annee = "2022";
let extension = "ano";
use pmsi_rust::core::parse_pmsi;
use pmsi_rust::core::utils::io::write_dataframe_to_somefile;

let ano = parse_pmsi(file_to_read.to_owned(), 
                     annee.to_owned(), 
                     extension.to_owned()).unwrap();
 
let out: String = "/Users/guillaumepressiat/Desktop/290000017.2022.12.ano.parquet".to_string();

 
write_dataframe_to_somefile(ano, out);
